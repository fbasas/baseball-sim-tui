def show_substitution_menu(self) -> None:
    """Show substitution menu modal.

    Gathers available pitchers and batters, marks used players,
    and pushes SubstitutionMenu modal.
    """
    if not self.away_team or not self.home_team:
        return

    state = self.game_state

    # Determine which team is fielding (for pitching change)
    if state.half == InningHalf.TOP:
        fielding_team = self.home_team
        batting_team = self.away_team
    else:
        fielding_team = self.away_team
        batting_team = self.home_team

    # Get current pitcher and batter
    current_pitcher_id = state.current_pitcher_id
    current_batter_slot = batting_team.lineup.get_batter(state.current_batting_index)
    current_batter_id = current_batter_slot.player_id

    # Build pitcher list (bullpen)
    pitchers = []
    for player in fielding_team.get_available_pitchers():
        if player.player_id == current_pitcher_id:
            continue  # Skip current pitcher
        stats = fielding_team.pitching_stats.get(player.player_id)
        era = stats.era if stats else 0.0
        available = self.sub_manager.is_player_available(player.player_id)
        pitchers.append((player.player_id, player.name_last, era, available))

    # Build batter list (bench)
    batters = []
    lineup_ids = [slot.player_id for slot in batting_team.lineup.slots]
    for player in batting_team.get_available_batters():
        if player.player_id in lineup_ids:
            continue  # Skip players already in lineup
        stats = batting_team.batting_stats.get(player.player_id)
        if stats and stats.at_bats > 0:
            avg = stats.hits / stats.at_bats
            # Calculate OBP and SLG for slash line
            obp = (stats.hits + stats.walks) / (stats.at_bats + stats.walks) if (stats.at_bats + stats.walks) > 0 else 0
            slg = (stats.hits + stats.doubles + 2*stats.triples + 3*stats.home_runs) / stats.at_bats if stats.at_bats > 0 else 0
            slash = f".{int(avg*1000):03d}/.{int(obp*1000):03d}/.{int(slg*1000):03d}"
        else:
            slash = ".000/.000/.000"
        available = self.sub_manager.is_player_available(player.player_id)
        batters.append((player.player_id, player.name_last, slash, available))

    # Push modal
    self.app.push_screen(
        SubstitutionMenu(
            pitchers=pitchers,
            batters=batters,
            current_pitcher_id=current_pitcher_id,
            current_batter_id=current_batter_id,
        ),
        self._handle_substitution
    )
```

def _handle_substitution(self, result: Optional[Tuple[str, str, str]]) -> None:
    """Process substitution menu result.

    Args:
        result: (sub_type, player_out, player_in) or None if cancelled
    """
    if result is None:
        return

    sub_type, player_out, player_in = result

    # Show confirmation dialog
    from textual.widgets import Button
    # For simplicity, proceed directly (confirmation deferred to polish)

    state = self.game_state

    if sub_type == "pitching_change":
        # Execute pitching change
        if state.half == InningHalf.TOP:
            # Home team fielding
            team = self.home_team
            new_state = state.with_pitcher(
                pitcher_id=player_in,
                fatigue=FatigueState(),
            )
            # Need to track which side
            new_state = dataclass_replace(new_state, home_pitcher_id=player_in, home_pitcher_fatigue=FatigueState())
        else:
            team = self.away_team
            new_state = dataclass_replace(state, away_pitcher_id=player_in, away_pitcher_fatigue=FatigueState())

        # Record in substitution manager
        record = SubstitutionRecord(
            inning=state.inning,
            half=state.half,
            sub_type=SubstitutionType.PITCHING_CHANGE,
            player_out_id=player_out,
            player_in_id=player_in,
            old_position=None,
            new_position=None,
            batting_order_slot=-1,  # Pitcher not in batting order with DH
        )
        self.sub_manager.record_substitution(record)

        # Update lineup's starting_pitcher_id
        team.lineup.starting_pitcher_id = player_in

        # Log to play log
        player_info = team.get_player(player_in)
        name = player_info.name_last if player_info else player_in
        log = self.query_one(PlayByPlayLog)
        log.add_play(f"[bold]Manager makes a pitching change: {name} now pitching[/bold]")

        self.game_state = new_state

    elif sub_type == "pinch_hitter":
        # Execute pinch hitter substitution
        if state.half == InningHalf.TOP:
            team = self.away_team
        else:
            team = self.home_team

        # Find slot in lineup
        slot_idx = state.current_batting_index % 9
        old_slot = team.lineup.slots[slot_idx]

        # Create new slot with replacement player
        from src.game.team import LineupSlot
        new_batting_stats = team.batting_stats[player_in]
        new_slot = LineupSlot(
            player_id=player_in,
            position=old_slot.position,  # Takes same position
            batting_stats=new_batting_stats,
        )
        team.lineup.slots[slot_idx] = new_slot

        # Record substitution
        record = SubstitutionRecord(
            inning=state.inning,
            half=state.half,
            sub_type=SubstitutionType.PINCH_HITTER,
            player_out_id=player_out,
            player_in_id=player_in,
            old_position=old_slot.position if isinstance(old_slot.position, Position) else None,
            new_position=old_slot.position if isinstance(old_slot.position, Position) else None,
            batting_order_slot=slot_idx,
        )
        self.sub_manager.record_substitution(record)

        # Log
        player_info = team.get_player(player_in)
        name = player_info.name_last if player_info else player_in
        log = self.query_one(PlayByPlayLog)
        log.add_play(f"[bold]Pinch hitter: {name} batting[/bold]")

        # Update lineup card
        self._update_lineup_cards()
```
